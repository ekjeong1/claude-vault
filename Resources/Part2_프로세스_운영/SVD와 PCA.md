from sklearn.preprocessing import StandardScaler

# ========================================

            label=f'80% threshold (PC{n_components_80})')

plt.ylabel('ì„¤ëª… ë¶„ì‚° ë¹„ìœ¨')
plt.title('Scree Plot')
plt.legend()
plt.grid(True)

plt.subplot(1, 2, 2)
plt.plot(range(1, len(cumulative_var)+1), cumulative_var, 'go-')

plt.xlabel('ì£¼ì„±ë¶„ ë²ˆí˜¸')
plt.ylabel('ëˆ„ì  ì„¤ëª… ë¶„ì‚°')
plt.title('ëˆ„ì  ë¶„ì‚°')
plt.legend()
plt.grid(True)
plt.tight_layout()
plt.savefig('pca_scree_plot.png', dpi=150)
print("\nScree plot ì €ì¥ ì™„ë£Œ: pca_scree_plot.png")

# ========================================
# 5. ì£¼ì„±ë¶„ í•´ì„ (ì ì¬ê°’ ë¶„ì„)
# ========================================
# ìµœì¢… ì£¼ì„±ë¶„ ê°œìˆ˜ ê²°ì • (80% ê¸°ì¤€ ì‚¬ìš©)

    pca.components_[:n_final].T,
    columns=[f'PC{i}' for i in range(1, n_final+1)],
    index=df.columns

    print(f"\n{col} (ì„¤ëª… ë¶„ì‚°: {explained_var[int(col[2:])-1]*100:.1f}%)")
    # ì ˆëŒ€ê°’ ê¸°ì¤€ ìƒìœ„ 5ê°œ KPI
    top5 = loadings[col].abs().nlargest(5)
    for kpi, loading in loadings.loc[top5.index, col].items():
        print(f"  {loading:+.3f}  {kpi}")

# ========================================
# 6. ì£¼ì„±ë¶„ ìŠ¤ì½”ì–´ ì €ì¥
# ========================================

    principal_components[:, :n_final],
    columns=[f'PC{i}' for i in range(1, n_final+1)],
    index=df.index

# CSV ì €ì¥ (ëŒ€ì‹œë³´ë“œì— ì‚¬ìš©)
pc_scores.to_csv('principal_component_scores.csv')
print(f"\nì£¼ì„±ë¶„ ìŠ¤ì½”ì–´ ì €ì¥ ì™„ë£Œ: principal_component_scores.csv")

# ========================================
# 7. ì‹œê°í™”: ì£¼ì„±ë¶„ ì‹œê³„ì—´
# ========================================
fig, axes = plt.subplots(n_final, 1, figsize=(12, 3*n_final))

    axes = [axes]

for i, ax in enumerate(axes, 1):
    pc_scores[f'PC{i}'].plot(ax=ax, color='steelblue', linewidth=2)
    ax.axhline(y=0, color='red', linestyle='--', alpha=0.5)
    ax.set_title(f'PC{i} ì‹œê³„ì—´ (ì„¤ëª… ë¶„ì‚°: {explained_var[i-1]*100:.1f}%)',
                 fontsize=12, fontweight='bold')
    ax.set_ylabel('í‘œì¤€í™” ìŠ¤ì½”ì–´')

# ========================================
# 8. ê²€ì¦: ì› ë°ì´í„° ë³µì›

print("\n" + "="*50)
print("PCA ë¶„ì„ ì™„ë£Œ!")
print("="*50)

def update_pc_scores(new_data_csv, scaler_file='scaler.pkl', pca_file='pca.pkl'):
    """
    ìƒˆë¡œìš´ ë°ì´í„°ê°€ ë“¤ì–´ì˜¤ë©´ ê¸°ì¡´ PCA ëª¨ë¸ë¡œ ì£¼ì„±ë¶„ ìŠ¤ì½”ì–´ ê³„ì‚°

    import pickle

    # ì €ì¥ëœ ëª¨ë¸ ë¡œë“œ

    with open(pca_file, 'rb') as f:

    # ìƒˆ ë°ì´í„° ë¡œë“œ
    new_df = pd.read_csv(new_data_csv, index_col='date', parse_dates=True)

    # ì „ì²˜ë¦¬ (ê°™ì€ scaler ì‚¬ìš©)

    new_pc_scores = pca.transform(new_scaled)

    # ê²°ê³¼ ë°˜í™˜
    result = pd.DataFrame(
        new_pc_scores,
        columns=[f'PC{i}' for i in range(1, pca.n_components_+1)],
        index=new_df.index
    )

    return result
